    1 import React, { useState } from 'react';
    2 import { ToastProvider } from './contexts/ToastContext';
    3 import MainSidebar from './components/layout/MainSidebar';
    4 import WorkingArea from './components/layout/WorkingArea';
    5 import Header from './components/layout/Header';
    6
    7 function App() {
    8   const [tabs, setTabs] = useState([{ id: 'welcome', title: 'Welcome', type: 'welcome' }]);
    9   const [activeTabId, setActiveTabId] = useState('welcome');
   10   const [categories, setCategories] = useState([
   11     { id: 1, name: 'General', notes: [{ id: 101, title: 'My First Note' }] },
   12     { id: 2, name: 'Work', notes: [] }
   13   ]);
   14
   15   const handleSelectTab = (tabId) => {
   16     setActiveTabId(tabId);
   17   };
   18
   19   const handleCloseTab = (tabIdToClose) => {
   20     // Prevent closing the 'welcome' tab
   21     if (tabIdToClose === 'welcome') {
   22       return;
   23     }
   24
   25     const tabIndex = tabs.findIndex(t => t.id === tabIdToClose);
   26     const newTabs = tabs.filter(t => t.id !== tabIdToClose);
   27     setTabs(newTabs);
   28
   29     // If the closed tab was the active one, set a new active tab
   30     if (activeTabId === tabIdToClose) {
   31       const newActiveTab = newTabs[tabIndex] || newTabs[tabIndex - 1] || newTabs[0];
   32       if (newActiveTab) {
   33         setActiveTabId(newActiveTab.id);
   34       } else {
   35         // This case should ideally not be reached if we can't close the welcome tab
   36         setActiveTabId(null);
   37       }
   38     }
   39   };
   40
   41   const openTodoListTab = () => {
   42     const todoTab = tabs.find(t => t.id === 'todo-list');
   43     if (todoTab) {
   44       setActiveTabId('todo-list');
   45     } else {
   46       const newTab = { id: 'todo-list', title: 'To-do List', type: 'todo' };
   47       setTabs([...tabs, newTab]);
   48       setActiveTabId(newTab.id);
   49     }
   50   };
   51
   52   const activeTab = tabs.find(t => t.id === activeTabId);
   53
   54   return (
   55     <ToastProvider>
   56       <div className="min-h-screen bg-slate-100 dark:bg-slate-900 text-slate-900 dark:text-slate-200 flex flex-col">
   57         <Header
   58           tabs={tabs}
   59           activeTabId={activeTabId}
   60           onSelectTab={handleSelectTab}
   61           onCloseTab={handleCloseTab}
   62           onOpenTodoList={openTodoListTab}
   63         />
   64         <div className="flex flex-row flex-grow">
   65           <MainSidebar categories={categories} />
   66           <WorkingArea activeTab={activeTab} />
   67         </div>
   68       </div>
   69     </ToastProvider>
   70   );
   71 }
   72
   73 export default App;
